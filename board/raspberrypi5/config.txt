# Enable hardware features

# GPU memory for video core (128 is enough for KMS + DSI + simple graphics)

# Enable Waveshare 7" DSI LCD (C)

# Some Waveshare panels work best with 2 framebuffers (helps with tearing)
max_framebuffers=2
